77 void ATCABoard::requestBus(void) {
78  buffer = 0x7b; //enable interrupt
79  i2c->sendData(ATCA_ARBITER, (char*)&buffer, 1, 0x05);
80  buffer = 0x00; //set request time
81  i2c->sendData(ATCA_ARBITER, (char*)&buffer, 1, 0x03);
82  buffer = 0x01; //request downstream
83  i2c->sendData(ATCA_ARBITER, (char*)&buffer, 1, 0x01);
84  if (checkAvailability()) {
85  buffer = 0x05; //enable downstream
86  i2c->sendData(ATCA_ARBITER, (char*)&buffer, 1, 0x01);
87  downstream_available = true;
88  } else {
89  printf("Bus not available.\n"); //TODO throw exception instead?
90  downstream_available = false;
91  exit(-1);
